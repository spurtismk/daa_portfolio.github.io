Restaurant Management System

System Overview

This Restaurant Management System is designed to streamline various operations in a restaurant. It handles customer queues, table reservations, inventory tracking, and order processing, aiming to optimize both customer experience and operational efficiency.

The key features of the system include:

  • Customer Queue Management
  • Table Reservation System
  • Inventory Management
  • Menu and Billing System
  • Delivery Management
  • Lodging and Room Booking

By integrating these systems into one cohesive platform, the Restaurant Management System ensures that tasks are handled smoothly, reducing wait times, optimizing table usage, and maintaining accurate records.

Customer Queue Management

The Customer Queue Management feature ensures that customers are served in the order they arrive. It uses a FIFO (First In, First Out) approach, guaranteeing that no customer is skipped or left waiting too long. This system helps in managing the customer flow and improving service efficiency.

Algorithm used:

  • FIFO (First In, First Out) algorithm

Table Reservation System

Our Table Reservation System allows customers to book tables in advance. This ensures that there is no confusion about table availability. The system keeps track of reserved and available tables, preventing double-booking and ensuring a smooth dining experience.

The table reservation algorithm checks if a table is available and reserves it for the customer. Once the customer is done, the table's status is updated as available again.

Algorithm used:

  • Table Lookup and Reservation Management

Code Implementation

C++ Code

#include <iostream>
#include <queue>
#include <map>
#include <string>

using namespace std;

// Customer Queue Management
class CustomerQueue {
    queue<string> customers;
public:
    void addCustomer(const string& name) {
        customers.push(name);
    }

    void removeCustomer() {
        if (!customers.empty()) {
            cout << "Customer " << customers.front() << " has been served.\\n";
            customers.pop();
        } else {
            cout << "The queue is empty. No customers to serve.\\n";
        }
    }

    void displayQueue() {
        if (customers.empty()) {
            cout << "No customers in the queue.\\n";
            return;
        }
        queue<string> temp = customers;
        cout << "Customers in the queue:\\n";
        while (!temp.empty()) {
            cout << temp.front() << "\\n";
            temp.pop();
        }
    }
};

// Table Reservation System
class Reservations {
    map<int, pair<string, int>> reservations;  // table number -> (customer name, number of seats)
    int totalTables;
public:
    Reservations(int tables) : totalTables(tables) {}

    bool reserveTable(int table, const string& name, int seats) {
        if (table > totalTables || table < 1) {
            cout << "Invalid table number.\\n";
            return false;
        }
        if (reservations.count(table)) {
            cout << "Table already reserved.\\n";
            return false;
        }
        reservations[table] = make_pair(name, seats);
        cout << "Table " << table << " reserved for " << name << " with " << seats << " seats.\\n";
        return true;
    }

    void releaseTable(int table) {
        if (reservations.erase(table)) {
            cout << "Table " << table << " is now available.\\n";
        } else {
            cout << "Table " << table << " is not reserved.\\n";
        }
    }

    void displayReservations() {
        if (reservations.empty()) {
            cout << "No reservations found.\\n";
            return;
        }
        cout << "Current reservations:\\n";
        for (auto &it : reservations) {
            cout << "Table " << it.first << ": " << it.second.first << " (Seats: " << it.second.second << ")\\n";
        }
    }
};

// Main function
int main() {
    CustomerQueue queue;
    Reservations reservations(10);  // 10 tables in the reservation system

    int choice;

                    cout << "4. Back\\n";
                    cout << "Enter your choice: ";
                    cin >> queueChoice;

                    if (queueChoice == 1) {
                        string name;
                        cout << "Enter customer name: ";
                        cin >> name;
                        queue.addCustomer(name);
                    } else if (queueChoice == 2) {
                        queue.removeCustomer();
                    } else if (queueChoice == 3) {
                        queue.displayQueue();
                    }
                } while (queueChoice != 4);
                break;
            }
            case 2: {
                int reservationChoice;
                do {
                    cout << "\\nReservation Management\\n";

                    cout << "4. Back\\n";
                    cout << "Enter your choice: ";

                    if (reservationChoice == 1) {

                        cin >> table;

                        cin >> name;

                        int table;
